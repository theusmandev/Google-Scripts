function createBloggerPosts() {
  // --- Settings ---
  var blogId = 'Your Blogger blog ID'; // Your Blogger blog ID
  var spreadsheetId = 'Your Google Sheet ID'; // Your Google Sheet ID
  var postLimit = 1; // Number of posts per run
  // ----------------

  var sheet = SpreadsheetApp.openById(spreadsheetId).getSheets()[0];
  var data = sheet.getDataRange().getValues();
  var count = 0;

  for (var i = 1; i < data.length; i++) {
    if (count >= postLimit) break;

    var title = data[i][0];
    var link = data[i][1];
    var status = data[i][2];

    if (status !== 'Done' && title && link) {
      try {
        // --- SEO-Friendly Description ---
        var seoDescription = `Read or download the popular Urdu novel '${title}' online for free. Explore the full story, complete chapters, and PDF links available for every Urdu literature enthusiast. This novel is perfect for readers who love emotional, and thrilling Urdu novels.
<br /><br />
Dive into the captivating world of ${title}, where the story unfolds with suspense, love, and unforgettable characters. Whether you want to read online on mobile or desktop or download the PDF for offline reading, this platform offers the best experience for Urdu novel lovers.

<p></p>
<div class="separator" style="clear: both; text-align: center;">
  <a href="https://blogger.googleusercontent.com/img/b/R29vZ2xl/AVvXsEh-bnKFZwp-OyvhxjNxr1bp5qawEEEHxNtP2lqZafMJ8b2_sREiYdPttx8IqA7JTTTSlIBxxZNmqCi34xZ3wPWiTUtAgq-xjh02SHTwfNbFu6FuhMQyciLB1Dot6K_jXyPTh_71kFKzhjeYELPVVk4Ue61e_aDa4xtV9G_8vhm7gVPiBG_ZemYTgDRpD0c/s125/urdunovelslinks.blogspot.com.gif" style="margin-left: 1em; margin-right: 1em;">
    <img alt="Download Urdu novels online - Urdu novels links for free reading" border="0" data-original-height="70" data-original-width="125" loading="lazy" src="https://blogger.googleusercontent.com/img/b/R29vZ2xl/AVvXsEh-bnKFZwp-OyvhxjNxr1bp5qawEEEHxNtP2lqZafMJ8b2_sREiYdPttx8IqA7JTTTSlIBxxZNmqCi34xZ3wPWiTUtAgq-xjh02SHTwfNbFu6FuhMQyciLB1Dot6K_jXyPTh_71kFKzhjeYELPVVk4Ue61e_aDa4xtV9G_8vhm7gVPiBG_ZemYTgDRpD0c/s16000/urdunovelslinks.blogspot.com.gif" title="Free Urdu novels download and read online - Urdu novels links" />
  </a>
</div>
<br />&nbsp;
<p></p>

`;

        // --- Card HTML with Header & Footer ---
        var postContent = `
<div class="novel-card">

  <!-- Header / Critical Content -->
  <div class="card-header">
    <h2 class="novel-title">${title}</h2>
    <p class="novel-description">${seoDescription}</p>

    <div class="novel-buttons">
      <a href="${link}" target="_blank" rel="noopener noreferrer" class="download-btn">Download PDF</a>
      <a href="${link}" target="_blank" rel="noopener noreferrer" class="read-btn">Read Online</a>
    </div>
  </div>

  <!-- Footer / Additional Content -->
  <div class="card-footer">
    <hr/>
    <h3>Why You Should Read ‘${title}’:</h3>
    <ul>
      <li>Complete Urdu novel chapters in one place.</li>
      <li>Easy online reading with user-friendly interface.</li>
      <li>Free PDF download to enjoy anytime, anywhere.</li>
      <li>Perfect for students, literary enthusiasts, and romance novel lovers.</li>
      <li>Ideal for those who follow famous Urdu writers and their latest works.</li>
    </ul>
<br />
    <p>Join thousands of readers who have already fallen in love with ${title}. Discover the twists, emotions, and drama that make this novel a must-read in Urdu literature.</p>
<br />
    <p><strong>Novel Tags/Keywords:</strong> Urdu novel, ${title} PDF, read Urdu novels online, free Urdu novel download, Urdu literature, Urdu novels online, Urdu story, emotional Urdu novel, popular Urdu novel.</p>
  </div>

</div>
        `;

        // --- Blogger API Call ---
        var url = 'https://www.googleapis.com/blogger/v3/blogs/' + blogId + '/posts';
        var payload = {
          "kind": "blogger#post",
          "blog": { "id": blogId },
          "title": title + " – Read & Download Urdu Novel Online",
          "content": postContent,
          "labels": ["Urdu Novel", "Urdu Novel PDFs"]
        };

        var options = {
          "method": "post",
          "contentType": "application/json",
          "headers": { "Authorization": "Bearer " + ScriptApp.getOAuthToken() },
          "payload": JSON.stringify(payload),
          "muteHttpExceptions": true
        };

        var response = UrlFetchApp.fetch(url, options);
        var responseCode = response.getResponseCode();

        if (responseCode === 200) {
          sheet.getRange(i + 1, 3).setValue('Done'); // Update status
          count++;
          Logger.log('Posted: ' + title);
          Utilities.sleep(10000); // 10 seconds pause to avoid rate limit
        } else {
          Logger.log('Failed: ' + title + ' - Response: ' + response.getContentText());
          if(responseCode == 429) { 
            Logger.log("Rate Limit Hit - Stopping for now.");
            break; 
          }
        }

      } catch (e) {
        Logger.log('Error posting ' + title + ': ' + e.toString());
      }
    }
  }
}
